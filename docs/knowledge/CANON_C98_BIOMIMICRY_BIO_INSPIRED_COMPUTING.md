# src/crystal_grid/slime_mould_router.py
"""
Slime mould-inspired routing for crystal grid telemetry
across damaged or disconnected network sections.
No centralized routing table. Paths emerge from local flow dynamics.
"""
import numpy as np
from typing import Dict, Tuple, List, Optional

class SlimeMouldRouter:
    """
    Models Physarum polycephalum tube-reinforcement dynamics.
    Tubes carrying more flow grow thicker; under-used tubes shrink.
    Constitutional use: route telemetry around failed sensor nodes.
    """

    def __init__(
        self,
        n_nodes: int,
        adjacency: np.ndarray,          # [N, N] connectivity matrix
        mu: float = 1.3,                # Flow-reinforcement exponent
        decay: float = 0.1,             # Tube decay per step
        dt: float = 0.01,
    ):
        self.n = n_nodes
        self.adj = adjacency
        self.mu = mu
        self.decay = decay
        self.dt = dt
        # Conductivity matrix (tube thickness — starts uniform)
        self.D = adjacency.copy().astype(float)
        self.D[self.D > 0] = 1.0

    def step(self, source: int, sink: int):
        """One Physarum update step: compute flows, update conductivities."""
        # Build Laplacian from conductivity
        L = np.zeros((self.n, self.n))
        for i in range(self.n):
            for j in range(self.n):
                if self.adj[i, j] > 0:
                    L[i, j] -= self.D[i, j]
                    L[i, i] += self.D[i, j]

        # Solve for pressure (Lp = b)
        b = np.zeros(self.n)
        b[source] = 1.0
        b[sink] = -1.0
        # Remove one equation for uniqueness
        L_red = L[:-1, :-1]
        b_red = b[:-1]
        try:
            p_red = np.linalg.solve(L_red, b_red)
        except np.linalg.LinAlgError:
            return  # Disconnected graph — no path
        p = np.append(p_red, 0.0)

        # Update conductivity: tubes that carry flow grow
        for i in range(self.n):
            for j in range(i + 1, self.n):
                if self.adj[i, j] > 0:
                    Q_ij = self.D[i, j] * abs(p[i] - p[j])
                    dD = Q_ij**self.mu / (1 + Q_ij**self.mu) - self.decay * self.D[i, j]
                    self.D[i, j] += self.dt * dD
                    self.D[j, i] = self.D[i, j]

    def find_path(self, source: int, sink: int, n_steps: int = 500) -> List[int]:
        """Run Physarum dynamics; return dominant path (highest conductivity)."""
        for _ in range(n_steps):
            self.step(source, sink)

        # Extract path by greedy highest-conductivity traversal
        path = [source]
        visited = {source}
        current = source
        while current != sink:
            neighbors = np.where(self.adj[current] > 0)[0]
            candidates = [n for n in neighbors if n not in visited]
            if not candidates:
                break
            next_node = max(candidates, key=lambda n: self.D[current, n])
            path.append(next_node)
            visited.add(next_node)
            current = next_node
        return path
